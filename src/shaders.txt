#shader vertex
attribute vec4 position;
varying vec4 fragColor;

uniform float pointSize;
uniform vec4 u_Color;
uniform mat4 u_MVP;

void main(void) {
    gl_Position =  u_MVP * position;
    gl_PointSize = pointSize;
    fragColor = u_Color;
}

#shader fragment
varying vec4 fragColor;

void main(void) {
    gl_FragColor = fragColor;
}
